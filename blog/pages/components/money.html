金额 - Money ​

￥ 21,890.23
我是一个 font-size 为 12px 的 div，我的钱包：
￥ 0.23
今日流水：
￥ 2,312,130.21
保留两位小数，67.325 =>：
￥ 67.33
vue
<script>
const FormattedMoney = computed(() => { // 文本展示
	const str = props.money.toLocaleString();
	let [, sign, trunc, , decimal] = /([-+]?)([0-9,]+)(\.(\d+))?/g.exec(str);
	decimal = parseFloat(`0.${decimal}`).toFixed(2);
	return sign + decimal.replace('0.', `${trunc}.`);
})

const TextedMoney = computed(() => { // tooltip 展示
	const str = props.money.toFixed(2).toString()
	if (props.money < 1000) {
		return `${str}元`
	}
	if (props.money < 10000) {
		return `${str[0]}千${parseInt(str.substring(1)) || ''}元`
	} else {
		const [ integer, decimal ] = str.split('.')
		const length = integer.toString().length
		return `${integer.substring(0, length-4)}万${parseInt(integer.substring(length-4, length-3)) ? `${integer.substring(length-4, length-3)}千`: ''}${parseInt(integer.substring(length-3)) || ''}${decimal ? `.${decimal}` : ''}元`
	}
})
</script>
